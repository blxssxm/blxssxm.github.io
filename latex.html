My template .tex

      
        %Document preamble*
        \documentclass[11pt]{article}
        \usepackage{geometry}
        \geometry{a4paper, left=20mm, right=20mm, top=10mm}
        
        %Relevant information*
        \title{}
        \author{}
        \date{\today}
        
        %Body Beginning*
        \begin{document}
        \maketitle
        \par\rule{\textwidth}{1pt}
        \tableofcontents
        
        %Adding sections
        \section*{}
        \subsection*{}
        \subsubsection*{}
        
        %Ending document*
        \end{document}
